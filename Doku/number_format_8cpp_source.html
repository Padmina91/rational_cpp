72 bool NumberFormat::check_string_correctness(const std::string& str, Format format) {
73  bool string_correct = true;
74  char thousand_sep = format == Format::german ? '.' : ',';
75  char decimal_sep = format == Format::german ? ',' : '.';
76  // first check if the spacing of thousand separators are correct, if applicable
77  size_t thousand_sep_at = str.find_first_of(thousand_sep);
78  size_t another_thousand_sep_at;
79  do {
80  if (thousand_sep_at != std::string::npos) {
81  another_thousand_sep_at = str.find_first_of(thousand_sep, thousand_sep_at + 1);
82  if (another_thousand_sep_at != std::string::npos) {
83  string_correct = another_thousand_sep_at - 4 == thousand_sep_at;
84  thousand_sep_at = another_thousand_sep_at;
85  }
86  }
87  } while (thousand_sep_at != std::string::npos && another_thousand_sep_at != std::string::npos);
88  // now thousand_sep_at has either the value npos or the position of the last thousand separator
89  // next check that that there are exactly 3 positions between the last thousand separator and the decimal separator
90  size_t decimal_sep_at = str.find_first_of(decimal_sep);
91  if (decimal_sep_at != std::string::npos && thousand_sep_at != std::string::npos && string_correct) {
92  string_correct = decimal_sep_at - 4 == thousand_sep_at;
93  }
94  // next make sure there is only one decimal separator in the string
95  if (str.find_first_of(decimal_sep, decimal_sep_at + 1) != std::string::npos) {
96  string_correct = false;
97  }
98  // next check if there are only numbers and separators in the string
99  if (str.find_first_not_of("-0123456789,.") != std::string::npos) {
100  string_correct = false;
101  }
102  // make sure that the algebraic sign '-' is either at the first position or non-existent
103  if (str.find_first_of('-') != 0 && str.find_first_of('-') != std::string::npos) {
104  string_correct = false;
105  }
106  return string_correct;
107 }

173 std::string NumberFormat::format_help(const Rational& x, Format format) {
174  char decimal_sep = format == Format::german ? ',' : '.';
175  double rational = x.double_value();
176  std::string return_string = std::to_string(rational);
177  if (format == Format::german) {
178  replace_character(return_string, '.', ",");
179  }
180  if (rational > 1000.0) {
181  add_thoudands_sep(return_string, format);
182  }
183  // remove unnecessary 0's at the end of string, if behind a decimal separator
184  if (return_string.find_first_of(decimal_sep) != std::string::npos && return_string.find_last_of('0') + 1 == return_string.length()) {
185  remove_unnecessary_zeros(return_string, decimal_sep);
186  }

198 Rational NumberFormat::parse_help(std::string& val_string, NumberFormat& nf) {
199  Format format = nf.get_format() == Format::english ? Format::english : Format::german;
200  bool correct_number = check_string_correctness(val_string, format);
201  if (!correct_number) {
202  throw NotANumberException();
203  }
204  if (format == Format::german) {
205  NumberFormat::remove_all(val_string, '.');
206  NumberFormat::replace_character(val_string, ',', ".");
207  } else {
208  NumberFormat::remove_all(val_string, ',');
209  }
210  double val_double = std::stod(val_string);
211  long long int denominator = 1;
212  while (static_cast<double>(static_cast<long long int>(val_double)) + 0.0000001 < val_double &&
213  static_cast<double>(static_cast<long long int>(val_double)) + 0.999999 > val_double) {
214  denominator *= 10;
215  val_double *= 10.0;
216  }
217  auto numerator = static_cast<long long int>(std::round(val_double));
218  Rational r(numerator, denominator, &nf);
219  return r;
220 }
